## 两数求和
```javascript
//给定一个整数数组 nums 和一个目标值 target，请你在该数组中找出和为目标值的那 两个 整数，并返回他们的数组下标。

//使用Map结构，将value值作为Map的key
function getTwoSum(arr, target) {
    const map = new Map()
    const combonation = []
    for(let [index, value] of arr.entries()) {
        map.set(value, index)
        if(map.has(target - value)) {
            combonation.push([map.get(target - value), index])
        }
    }
    return combonation
}
```

### 合并有序数组
```javascript
// 两个有序数组arr1和arr2，假设arr1数组空间足够大，请把arr2合并到arr1
function merge(arr1, arr2) {
    let arr1Index = arr1.length - 1
    let arr2Index = arr2.length - 1
    let sumIndex = arr1Index + arr2Index + 1
    while(arr1Index>=0 && arr2Index>=0) {
        if(arr1[arr1Index]>arr2[arr2Index]) {
            arr1[sumIndex] = arr1[arr1Index]
            arr1Index--
        }else {
            arr1[sumIndex] = arr2[arr2Index]
            arr2Index--
        }
        sumIndex--
    }

    while(arr2Index>=0) {
        arr1[sumIndex] = arr2[arr2Index]
        sumIndex--
        arr2Index--
    }
    return arr1
}
```
