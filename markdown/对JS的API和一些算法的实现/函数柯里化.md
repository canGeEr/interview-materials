# 函数柯里化
```javascript
//用于为一个函数绑定一个或者多个参数
const curry = (function() {
    let args = []
    let callback = null
    let context = null
    function curried(argument) {
        args.push(argument)
        if(args.length < callback.length) {
            return curried.bind(context)
        }else {
            let result = callback.apply(context, args)
            args = [] //希望用完下次还能用
            return result
        }
    }

    return function(cb, ct) {
        callback = cb 
        context = ct
        return curried.bind(context)
    }
})()

function fun(a, b, c) {
    return a+b+c
}

let newFun = curry(fun)
```
