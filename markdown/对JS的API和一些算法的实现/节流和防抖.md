## 节流
```javascript
function throttle(callback, delay) {
    let timer = null
    return function(...args) {
        if(!timer) {
            setTimeout(()=>{
                callback.apply(this, args)
                timer = null
            }, delay)
        }   
    }
}
```

## 防抖
```javascript
function debounce(callback, delay) {
    let timer = null
    return function(...args) {
        if(timer) clearTimeout(timer)
        timer = setTimeout(()=>{
            callback.apply(this, args)
        }, delay)
    }
}
```
