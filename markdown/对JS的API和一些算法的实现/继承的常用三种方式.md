## 组合式继承
```javascript
function Parent(...args) {
    //调用this绑定实例上的属性和方法
}

function Child(...args) {
    //对父亲构造函数的调用是点睛之笔
    Parent.apply(this, args) // Parent.call(this, ...args) 
    //调用this绑定实例上的属性和方法
}

Child.prototype = new Parent()
```

## 组合寄生式继承
```javascript
function Parent(...args) {
}

function Child(...args) {
    Parent.apply(this, args)
}

Child.prototype = Object.create(Parent.prototype, {
    constructor: {
        value: Child,
        wirtable: true,
        enumerable: false,
        configurable: true
    }
})
```
