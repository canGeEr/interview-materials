class EventBus {
  dep
  constructor() {
    this.dep = {}
  }
  on(eventName, callback) {
    if(this.dep[eventName]) {
      this.dep[eventName] = []
    }
    this.dep[eventName].push(callback)
  }
  emit(eventName, ...args) {
    const callbackArr = this.dep[eventName]
    callbackArr && (callbackArr.forEach(callback => {
      callback(...args)
    }))
  }

  }

    if(this.dep[eventName]) {
