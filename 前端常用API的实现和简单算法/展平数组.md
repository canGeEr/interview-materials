## 递归
```javascript
function flat(arr) {
    let result = []
    for(let value of arr) {
        if(Array.isArray(value)) {
            result = result.concat(flat(value))
        }else {
            result.push(value)
        }
    }
    return result
}

//reduce机制去简写
function flat(arr) {
    return arr.reduce(function (prev, item) {
        return prev.concat(Array.isArray(item) ? flat(item) : [item]);
    }, []);
}

function flat(arr){
    while(arr.some(item=>Array.isArray(item))){
        arr = [].concat(...arr);
    }
    return arr;
}
```
